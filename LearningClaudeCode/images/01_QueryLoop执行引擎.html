# ── 启动阶段 ─────────────────────
def startup():
    load_claude_md()        # 项目记忆
    load_memory_files()     # 持久记忆
    build_system_prompt()   # 组装提示
    connect_mcp_servers()   # MCP 初始化

# ── 主循环 ───────────────────────
def query_loop(user_input):
    messages = [system_prompt, user_input]

    while True:
        # ① REASONING
        response = claude_api(messages)

        if response.has_tool_call:
            tc = response.tool_calls[0]

            # ② PERMISSION CHECK
            perm = check_permissions(tc)
            if perm == DENY: continue

            # ③ PRE HOOK
            run_hooks("PreToolUse", tc)

            # ④ ACTION
            result = execute_tool(tc)

            # ⑤ POST HOOK
            run_hooks("PostToolUse", tc, result)

            # ⑥ INJECT
            messages.append(result)

            # ⑦ VALIDATION → compress?
            if ctx_ratio() > 0.70:
                compress_context(messages)

        else:
            # FINAL OUTPUT
            display_output(response.text)
            run_hooks("Stop", response)
            user_input = wait_input()
            if not user_input: break
